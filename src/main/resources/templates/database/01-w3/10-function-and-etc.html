함수

trim

SELECT TRIM('  Hello World!  '); -- 'Hello World!'
SELECT TRIM(LEADING ' ' FROM '  Hello World!  '); -- 'Hello World!  ' 
SELECT TRIM(TRAILING '*' FROM '  Hello World!****'); -- '  Hello World!'

concat

SELECT CONCAT('Hello', ' ', 'World!'); -- 'Hello World!'
SELECT CONCAT('Hello', ' ', 'World!', 123); -- 'Hello World!123'

concat_ws

SELECT CONCAT_WS('-', '2023', '10', '01'); -- '2023-10-01'

substring

SELECT SUBSTRING('Hello World!', 1, 5); -- 'Hello'

replace

SELECT REPLACE('Hello World!', 'World', 'MySQL'); -- 'Hello MySQL!'

upper

SELECT UPPER('Hello World!'); -- 'HELLO WORLD!'

lower

SELECT LOWER('Hello World!'); -- 'hello world!'

length

SELECT LENGTH('Hello World!'); -- 13

if

SELECT IF(1 > 0, 'True', 'False'); -- 'True'
SELECT IF(1 > 2, 'True', 'False'); -- 'False'

case

SELECT CASE 
      WHEN 1 > 0 THEN 'True' 
      ELSE 'False' 
  END; -- 'True'
SELECT CASE 
      WHEN 1 > 2 THEN 'True' 
      ELSE 'False' 
  END; -- 'False'

coalesce

SELECT COALESCE(NULL, 'Hello', 'World'); -- 'Hello'
SELECT COALESCE(NULL, NULL, 'World'); -- 'World'
SELECT COALESCE(NULL, NULL, NULL); -- NULL

ifnull; 첫번째가 null이면 다음값, 아니면 그냥 값

SELECT IFNULL(NULL, 'Hello'); -- 'Hello'
SELECT IFNULL('Hello', 'World'); -- 'Hello'

nullif: 2개의 값이 같으면 null, 아니면 첫번째 값

SELECT NULLIF('Hello', 'Hello'); -- NULL
SELECT NULLIF('Hello', 'World'); -- 'Hello'

greatest; 여러 값 중 가장 큰 값

SELECT GREATEST(1, 2, 3); -- 3

least; 여러 값 중 가장 작은 값

SELECT LEAST(1, 2, 3); -- 1

수학 함수

format; 숫자를 지정된 형식으로 포맷

SELECT FORMAT(1234567.89, 2); -- '1,234,567.89'

round; 숫자를 반올림

SELECT ROUND(123.4567, 2); -- 123.46
SELECT ROUND(123.4567); -- 123

ceil; 숫자를 올림

SELECT CEIL(123.4567); -- 124
SELECT CEIL(-123.4567); -- -123

floor; 숫자를 내림

SELECT FLOOR(123.4567); -- 123
SELECT FLOOR(-123.4567); -- -124

abs; 절대값

SELECT ABS(-123.4567); -- 123.4567
SELECT ABS(123.4567); -- 123.4567

sign; 숫자의 부호 반환

SELECT SIGN(-123.4567); -- -1
SELECT SIGN(0); -- 0
SELECT SIGN(123.4567); -- 1

rand; 랜덤 숫자 생성

SELECT RAND(); -- 0과 1 사이의 랜덤 숫자
SELECT RAND(123); -- 123을 시드로 사용하여 랜덤 숫자 생성

pi; 원주율 반환

SELECT PI(); -- 3.141592653589793

mod; 나머지 연산

SELECT MOD(10, 3); -- 1
SELECT MOD(10, 2); -- 0

시스템 함수

version; MySQL 버전 반환

SELECT VERSION(); -- MySQL 서버 버전 반환

user; 현재 사용자 반환

SELECT USER(); -- 현재 사용자 반환

current_user; 현재 사용자 반환

SELECT CURRENT_USER(); -- 현재 사용자 반환

database; 현재 데이터베이스 반환

SELECT DATABASE(); -- 현재 데이터베이스 반환

schema; 현재 스키마 반환

SELECT SCHEMA(); -- 현재 스키마 반환

DATE, TIME

now

SELECT NOW(); -- 현재 날짜와 시간

curdate

SELECT CURDATE(); -- 현재 날짜
SELECT CURRENT_DATE(); -- 현재 날짜 반환
SELECT CURRENT_DATE; -- 현재 날짜 반환

curtime

SELECT CURTIME(); -- 현재 시간
SELECT CURRENT_TIME(); -- 현재 시간 반환
SELECT CURRENT_TIME; -- 현재 시간 반환

date_format

SELECT DATE_FORMAT(NOW(), '%Y-%m-%d %H:%i:%s'); -- '2023-10-01 12:34:56'

current_timestamp; 현재 날짜와 시간 반환

SELECT CURRENT_TIMESTAMP(); -- 현재 날짜와 시간 반환
SELECT CURRENT_TIMESTAMP; -- 현재 날짜와 시간 반환
SELECT TIMESTAMP(NOW()); -- 현재 날짜와 시간 반환

sysdate; 현재 날짜와 시간 반환

SELECT SYSDATE(); -- 현재 날짜와 시간 반환
SELECT SYSDATE; -- 현재 날짜와 시간 반환

날짜 관련 데이터 추출하기

SELECT DATE('2023-10-01 12:34:56') / TIME('2023-10-01 12:34:56') 
  / YEAR('2023-10-01 12:34:56') / MONTH('2023-10-01 12:34:56') / DAY('2023-10-01 12:34:56') 
  / HOUR('2023-10-01 12:34:56') / MINUTE('2023-10-01 12:34:56') / SECOND('2023-10-01 12:34:56');
   -- '2023-10-01' / '12:34:56' / 2023 / 10 / 1 / 12 / 34 / 56

SELECT DATE(NOW()) / TIME(NOW()) / YEAR(NOW()) / MONTH(NOW()) / DAY(NOW()) 
  / HOUR(NOW()) / MINUTE(NOW()) / SECOND(NOW()); 
  -- 현재 날짜, 시간, 년, 월, 일, 시, 분, 초 반환

extract; 날짜에서 특정 부분 추출

SELECT EXTRACT(YEAR FROM '2023-10-01'); -- 2023
SELECT EXTRACT(MONTH FROM '2023-10-01'); -- 10
SELECT EXTRACT(DAY FROM '2023-10-01'); -- 1

date_add; 날짜에 시간 더하기

SELECT DATE_ADD('2023-10-01', INTERVAL 1 DAY); -- '2023-10-02'
SELECT DATE_ADD('2023-10-01', INTERVAL 1 MONTH); -- '2023-11-01'

date_sub; 날짜에서 시간 빼기

SELECT DATE_SUB('2023-10-01', INTERVAL 1 DAY); -- '2023-09-30'
SELECT DATE_SUB('2023-10-01', INTERVAL 1 MONTH); -- '2023-09-01'

date_diff; 날짜 차이

SELECT DATEDIFF('2023-10-01', '2023-09-01'); -- 30
SELECT DATEDIFF('2023-10-01 12:00:00', '2023-09-01 12:00:00'); -- 30

timestampdiff; 두 날짜 사이의 차이를 특정 단위로 반환

SELECT TIMESTAMPDIFF(DAY, '2023-10-01', '2023-09-01'); -- -30
SELECT TIMESTAMPDIFF(MONTH, '2023-10-01', '2023-09-01'); -- -1

unix_timestamp; 현재 시간을 유닉스 타임스탬프로 반환

SELECT UNIX_TIMESTAMP(); -- 현재 시간을 유닉스 타임스탬프로 반환
SELECT UNIX_TIMESTAMP('2023-10-01 12:00:00'); -- 특정 시간을 유닉스 타임스탬프로 반환

from_unixtime; 유닉스 타임스탬프를 날짜로 변환

SELECT FROM_UNIXTIME(UNIX_TIMESTAMP()); -- 현재 시간을 날짜로 변환
SELECT FROM_UNIXTIME(1696129200); -- 특정 유닉스 타임스탬프를 날짜로 변환

utc_date; UTC 날짜 반환

SELECT UTC_DATE(); -- UTC 날짜 반환

utc_time; UTC 시간 반환

SELECT UTC_TIME(); -- UTC 시간 반환
